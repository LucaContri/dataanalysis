  PRIMARY KEY (`name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=COMPACT;

insert into countries_names 
	select `name`, `code` from analytics.countries
	union 
	select `full_name`, `code` from analytics.countries
	union 
	select `code`, `code` from analytics.countries
	union 
	select `iso3`, `code` from analytics.countries;
